Implementation — How Each Service Handles Correlation IDs
☕ Spring Boot — CorrelationIdFilter.java
@Component @Order(1) public class CorrelationIdFilter extends OncePerRequestFilter { static final String HEADER = "X-Correlation-Id"; @Override protected void doFilterInternal( HttpServletRequest req, HttpServletResponse res, FilterChain chain) throws ... { String cid = req.getHeader(HEADER); if (cid == null) cid = UUID.randomUUID().toString(); MDC.put("correlationId", cid); res.setHeader(HEADER, cid); try { chain.doFilter(req, res); } finally { MDC.clear(); } } }
☕ Spring Boot — logback-spring.xml
<!-- logback-spring.xml --> <appender name="CONSOLE" class="ConsoleAppender"> <encoder class="LogstashEncoder"> <customFields> {"service":"${appName}"} </customFields> <!-- pulls from MDC automatically --> <includeMdcKeyName> correlationId </includeMdcKeyName> </encoder> </appender>
🐍 FastAPI — correlation middleware
# main.py (user-service, catalog-service) @app.middleware("http") async def correlation_id( request: Request, call_next): cid = request.headers.get( "X-Correlation-Id", str(uuid.uuid4()) # auto-generate ) response = await call_next(request) response.headers["X-Correlation-Id"] = cid logger.info("request", extra={ "cid": cid, "path": request.url.path, "method": request.method, "status": response.status_code, }) return response
